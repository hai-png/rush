   168→//   2. RSA-PSS-SHA256 signature in the request body's `sign` field
   169→//   - SHA256withRSA with PSS padding + MGF1-SHA256 (NOT PKCS#1 v1.5)
   170→//   - Exclude fields: sign, sign_type, header, refund_info, openType,
   171→//   - Sort keys lexicographically, join as k=v&k=v
   172→class TelebirrProvider implements PaymentProvider {
   173→  readonly name = 'telebirr' as const;
   174→  private env = loadEnv();
   175→
   176→  private get apiBase(): string {
   177→    return this.env.TELEBIRR_ENV === 'production'
   178→      ? 'https://superapp.ethiomobilemoney.et:38443/apiaccess/payment/gateway'
   179→      : 'https://developerportal.ethiotelebirr.et:38443/apiaccess/payment/gateway';
   180→  }
   181→
   182→  private get webBase(): string {
   183→    // Testbed ends with `?` (the URL builder appends `&field=...`).
   184→    // Production: confirm with Ethio telecom; using the documented testbed shape.
   185→    return this.env.TELEBIRR_ENV === 'production'
   186→      ? 'https://superapp.ethiomobilemoney.et:38443/payment/web/paygate?'
   187→      : 'https://developerportal.ethiotelebirr.et:38443/payment/web/paygate?';
   188→  }
   189→
   190→  // Excluded from signing per Telebirr docs. Note: `biz_content` is excluded
   191→  // as a key but its children ARE flattened and signed.
   192→  private static EXCLUDE_FIELDS = new Set([
   193→    'sign', 'sign_type', 'header', 'refund_info',
   194→    'openType', 'raw_request', 'biz_content', 'wallet_reference_data',
   195→  ]);
   196→
   197→  private buildStringToSign(req: Record<string, any>): string {
   198→    const flat: Record<string, string> = {};
   199→    for (const [k, v] of Object.entries(req)) {
   200→      if (TelebirrProvider.EXCLUDE_FIELDS.has(k)) continue;
   201→      if (v === undefined || v === null || v === '') continue;
   202→      flat[k] = String(v);
   203→    }
   204→    // biz_content is excluded as a key, but its children are flattened + signed.
   205→    const biz = req.biz_content;
   206→    if (biz && typeof biz === 'object') {
   207→      for (const [k, v] of Object.entries(biz)) {
   208→        if (TelebirrProvider.EXCLUDE_FIELDS.has(k)) continue;
   209→        if (v === undefined || v === null || v === '') continue;
   210→        flat[k] = String(v);
   211→      }
   212→    }
   213→    return Object.keys(flat)
   214→      .sort()
   215→      .map((k) => `${k}=${flat[k]}`)
   216→      .join('&');
   217→  }
   218→
   219→  private signTelebirr(req: Record<string, any>): string {
   220→    const data = this.buildStringToSign(req);
   221→    return createSign('sha256')
   222→      .update(data, 'utf8')
   223→      .sign({
   224→        key: this.env.TELEBIRR_PRIVATE_KEY,
   225→        padding: 1, // RSA_PKCS1_PSS_PADDING
   226→        saltLength: 0, // RSA_PSS_SALTLEN_DIGEST
   227→      })
   228→      .toString('base64');
   229→  }
   230→
   231→  private verifyTelebirr(payload: Record<string, any>, signatureBase64: string): boolean {
   232→    const data = this.buildStringToSign(payload);
   233→    try {
   234→      return createVerify('sha256')
   235→        .update(data, 'utf8')
   236→        .verify({
   237→          key: this.env.TELEBIRR_PUBLIC_KEY,
   238→          padding: 1, // RSA_PKCS1_PSS_PADDING
   239→          saltLength: 0,
   240→        }, signatureBase64, 'base64');
   241→    } catch {
   242→      return false;
   243→    }
   244→  }
   245→
   246→  private createNonceStr(): string {
   247→    return randomUUID().replace(/-/g, '');
   248→  }
   249→
   250→  private createTimestamp(): string {
   251→    return Math.floor(Date.now() / 1000).toString();
   252→  }
   253→
   254→  private cachedToken: { token: string; expiresAt: number } | null = null;
   255→
   256→  private async applyFabricToken(): Promise<string> {
   257→    if (this.cachedToken && Date.now() < this.cachedToken.expiresAt - 60_000) {
   258→      return this.cachedToken.token;
   259→    }
   260→    const res = await fetch(`${this.apiBase}/payment/v1/token`, {
   261→      method: 'POST',
   262→      headers: {
   263→        'Content-Type': 'application/json',
   264→        'X-APP-Key': this.env.TELEBIRR_FABRIC_APP_ID,
   265→      },
   266→      body: JSON.stringify({ appSecret: this.env.TELEBIRR_APP_SECRET }),
   267→    });
   268→    if (!res.ok) {
   269→      throw new Error(`applyFabricToken HTTP ${res.status}: ${await res.text()}`);
   270→    }
   271→    const data = await res.json() as { token: string; expirationDate?: string };
   272→    // expirationDate is yyyyMMddHHmmss; fall back to 1h if absent.
   273→    const expiresAt = data.expirationDate ? this.parseTelebirrDate(data.expirationDate) : Date.now() + 3600_000;
   274→    this.cachedToken = { token: data.token, expiresAt };
   275→    return data.token;
   276→  }
   277→
   278→  private parseTelebirrDate(s: string): number {
   279→    // yyyyMMddHHmmss -> epoch ms
   280→    const y = +s.slice(0, 4);
   281→    const mo = +s.slice(4, 6) - 1;
   282→    const d = +s.slice(6, 8);
   283→    const h = +s.slice(8, 10);
   284→    const mi = +s.slice(10, 12);
   285→    const se = +s.slice(12, 14);
   286→    return Date.UTC(y, mo, d, h, mi, se);
   287→  }

   289→  private async callBusinessApi<T = any>(
   290→    endpoint: string,
   291→    bizContent: Record<string, any>,
   292→    method: string,
   293→  ): Promise<T> {
   294→    const token = await this.applyFabricToken();
   295→    const req: Record<string, any> = {
   296→      timestamp: this.createTimestamp(),
   297→      nonce_str: this.createNonceStr(),
   298→      method,
   299→      version: '1.0',
   300→      biz_content: bizContent,
   301→    };
   302→    req.sign = this.signTelebirr(req);
   303→    req.sign_type = 'SHA256WithRSA';
   304→
   305→    const res = await fetch(`${this.apiBase}${endpoint}`, {
   306→      method: 'POST',
   307→      headers: {
   308→        'Content-Type': 'application/json',
   309→        'X-APP-Key': this.env.TELEBIRR_FABRIC_APP_ID,
   310→        'Authorization': token, // already includes 'Bearer '
   311→      },
   312→      body: JSON.stringify(req),
   313→    });
   314→    if (!res.ok) {
   315→      throw new Error(`Telebirr ${endpoint} HTTP ${res.status}: ${await res.text()}`);
   316→    }
   317→    return res.json() as Promise<T>;
   318→  }

   320→  async createCheckout(intent: PaymentIntent): Promise<CheckoutResult> {
   321→    const bizContent = {
   322→      appid: this.env.TELEBIRR_MERCHANT_APP_ID,
   323→      merch_code: this.env.TELEBIRR_MERCHANT_CODE,
   324→      merch_order_id: intent.merchOrderId,
   325→      title: intent.description,
   326→      total_amount: intent.amount.toDecimalString(),
   327→      trans_currency: 'ETB',
   328→      timeout_express: '120m',
   329→      notify_url: intent.notifyUrl,
   330→      redirect_url: intent.redirectUrl,
   331→      payee_type: '5000', // H5 sample value; confirm with ET
   332→    };
   333→
   334→    const response = await this.callBusinessApi<{
   335→      result: string;
   336→      code: string;
   337→      msg: string;
   338→      biz_content: { prepay_id: string; merch_order_id: string };
   339→      sign: string;
   340→    }>('/payment/v1/merchant/preOrder', bizContent, 'payment.preorder');
   341→
   342→    if (response.result !== 'SUCCESS' || response.code !== '0') {
   343→      throw new Error(`Telebirr preOrder failed: ${response.msg} (code ${response.code})`);
   344→    }
   345→
   346→    const checkoutUrl = this.buildCheckoutUrl(response.biz_content.prepay_id, response.sign);
   347→    return {
   348→      status: 'checkout',
   349→      checkoutUrl,
   350→      prepayId: response.biz_content.prepay_id,
   351→    };
   352→  }

   354→  // Build the user-facing redirect URL: web/paygate?<signed fields>.
   355→  private buildCheckoutUrl(prepayId: string, sign: string): string {
   356→    // Per the docs, the checkout URL is the webBase + a sorted query string
   357→    // of the biz_content fields + sign + sign_type, then &version=1.0&trade_type=Checkout.
   358→    const timestamp = this.createTimestamp();
   359→    const nonceStr = this.createNonceStr();
   360→    const queryString = [
   361→      `appid=${this.env.TELEBIRR_MERCHANT_APP_ID}`,
   362→      `merch_code=${this.env.TELEBIRR_MERCHANT_CODE}`,
   363→      `nonce_str=${nonceStr}`,
   364→      `prepay_id=${prepayId}`,
   365→      `timestamp=${timestamp}`,
   366→      `sign=${sign}`,
   367→      `sign_type=SHA256WithRSA`,
   368→    ].join('&');
   369→    return `${this.webBase}${queryString}&version=1.0&trade_type=Checkout`;
   370→  }

   417→  async verifyPayment(reference: string): Promise<PaymentStatusResult> {
   418→    const bizContent = {
   419→      appid: this.env.TELEBIRR_MERCHANT_APP_ID,
   420→      merch_code: this.env.TELEBIRR_MERCHANT_CODE,
   421→      merch_order_id: reference,
   422→    };
   423→    const response = await this.callBusinessApi<{
   424→      result: string;
   425→      code: string;
   426→      msg: string;
   427→      biz_content: {
   428→        trade_status: string;
   429→        total_amount?: string;
   430→        trans_id?: string;
   431→      };
   432→    }>('/payment/v1/merchant/queryOrder', bizContent, 'payment.query');
   433→
   434→    if (response.result !== 'SUCCESS') {
   435→      return { status: 'pending', raw: response };
   436→    }
   437→
   438→    // queryOrder trade_status: WAIT_PAY, PAYING, PAY_SUCCESS, PAY_FAILED,
   439→    // ORDER_CLOSED, ACCEPTED, REFUNDING, REFUND_SUCCESS, REFUND_FAILED
   440→    let status: 'pending' | 'completed' | 'failed' = 'pending';
   441→    if (response.biz_content.trade_status === 'PAY_SUCCESS') status = 'completed';
   442→    else if (response.biz_content.trade_status === 'PAY_FAILED' || response.biz_content.trade_status === 'ORDER_CLOSED') status = 'failed';
   443→
   444→    const amount = response.biz_content.total_amount
   445→      ? Money.fromETBString(response.biz_content.total_amount)
   446→      : undefined;
   447→    return { status, amount, raw: response };
   448→  }

   450→  async refund(req: RefundRequest): Promise<RefundResult> {
   451→    const bizContent = {
   452→      appid: this.env.TELEBIRR_MERCHANT_APP_ID,
   453→      merch_code: this.env.TELEBIRR_MERCHANT_CODE,
   454→      merch_order_id: req.merchOrderId,
   455→      refund_request_no: req.refundRequestNo,
   456→      refund_reason: req.reason,
   457→      actual_amount: req.amount.toDecimalString(),
   458→      trans_currency: 'ETB',
   459→    };
   460→    try {
   461→      const response = await this.callBusinessApi<{
   462→        result: string;
   463→        code: string;
   464→        msg: string;
   465→        biz_content: {
   466→          refund_status: string;
   467→          refund_amount?: string;
   468→        };
   469→      }>('/payment/v1/merchant/refund', bizContent, 'payment.refund');
   470→
   471→      if (response.result !== 'SUCCESS') {
   472→        return { status: 'failed', error: response.msg || 'Unknown error', permanent: false };
   473→      }
   474→
   475→      const st = response.biz_content.refund_status;
   476→      if (st === 'REFUND_SUCCESS' || st === 'REFUND_DUPLICATED') return { status: 'succeeded' };
   477→      if (st === 'REFUNDING') return { status: 'processing', retryAfterMs: 15 * 60_000 };
   478→      return {
   479→        status: 'failed',
   480→        error: `Refund ${st}`,
   481→        permanent: st === 'REFUND_FAILED',
   482→      };
   483→    } catch (e) {
   484→      return { status: 'failed', error: (e as Error).message, permanent: false };
   485→    }
   486→  }

   488→  // not a backend HTTP API. We just build the URL with the merchant's params.

   558→      trans_currency: 'ETB',

   560→    };

   562→      const response = await this.callBusinessApi<{
   563→        result: string;
   564→        code: string;
   565→        msg: string;
   566→        biz_content: {
   567→          payment_order_id?: string;
   568→          disburse_status?: string; // SUCCESS, PROCESSING, FAILED
   569→        };
   570→      }>('/payment/v1/merchant/disburseOrder', bizContent, 'payment.disbursement');
   571→
   572→      if (response.result !== 'SUCCESS') {
   573→        return { status: 'failed', error: response.msg || 'Unknown error', permanent: false };
   574→      }
   575→      const st = (response.biz_content.disburse_status ?? '').toUpperCase();
   576→      const paymentOrderId = response.biz_content.payment_order_id ?? '';
   577→      if (st === 'SUCCESS') return { status: 'succeeded', paymentOrderId };
   578→      if (st === 'PROCESSING') return { status: 'processing', paymentOrderId };
   579→      return { status: 'failed', error: `Disburse ${st}`, permanent: st === 'FAILED' };
   580→    } catch (e) {
   581→      return { status: 'failed', error: (e as Error).message, permanent: false };
   582→    }

   585→  async parseWebhook(req: Request): Promise<WebhookEvent> {
   586→    const raw = await req.text();
   587→    let payload: any;
   588→    try { payload = JSON.parse(raw); } catch { throw new Error('Invalid telebirr webhook JSON'); }
   589→
   590→    // Timestamp: docs say "13 chars" but samples use 10-digit Unix seconds.
   591→    // Accept either; reject if >5min old (replay protection).
   592→    const ts = typeof payload.timestamp === 'number'
   593→      ? payload.timestamp
   594→      : typeof payload.timestamp === 'string' && payload.timestamp.trim() !== '' && !isNaN(Number(payload.timestamp))
   595→        ? Number(payload.timestamp)
   596→        : undefined;
   597→    if (ts === undefined) {
   598→      throw new Error('Telebirr webhook missing numeric timestamp — possible replay');
   599→    }
   600→    const timestampMs = ts < 1e12 ? ts * 1000 : ts; // seconds -> ms
   601→    if (Date.now() - timestampMs > 5 * 60_000) {
   602→      throw new Error('Telebirr webhook timestamp too old (replay suspected)');
   603→    }
   604→
   605→    // Verify signature using SP's public key.
   606→    let signatureValid = false;
   607→    if (typeof payload.sign === 'string' && payload.sign && this.env.TELEBIRR_PUBLIC_KEY) {
   608→      signatureValid = this.verifyTelebirr(payload, payload.sign);
   609→    }
   610→
   611→    // Notify trade_status values: Completed, Failure, Pending, Paying, Expired
   612→    // (different from queryOrder's PAY_SUCCESS/PAY_FAILED)
   613→    const outRequestNo = payload.out_request_no ?? payload.trans_id ?? `orno-${Date.now()}`;
   614→
   615→    if (payload.refund_request_no) {
   616→      // Refund notify (rare — refunds usually sync via refund() response)
   617→      return payload.trade_status === 'Completed' || payload.trade_status === 'REFUND_SUCCESS'
   618→        ? { type: 'refund.succeeded', refundRequestNo: payload.refund_request_no, raw: payload, signatureValid, timestampMs }
   619→        : { type: 'refund.failed', refundRequestNo: payload.refund_request_no, raw: payload, signatureValid, timestampMs };
   620→    }
   621→
   622→    return payload.trade_status === 'Completed' || payload.trade_status === 'Success' || payload.trade_status === 'PAY_SUCCESS'
   623→      ? { type: 'payment.settled', merchOrderId: payload.merch_order_id, amount: Money.fromETBString(payload.total_amount), raw: payload, signatureValid, timestampMs, outRequestNo, transId: payload.trans_id }
   624→      : { type: 'payment.failed', merchOrderId: payload.merch_order_id, raw: payload, signatureValid, timestampMs, outRequestNo };
   625→  }
   626→}
